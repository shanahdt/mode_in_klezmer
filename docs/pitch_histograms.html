  • Observable

data = FileAttachment("freygish_1.csv").csv({typed: true})
Plot.plot({
  marginBottom: 100,
  y: {grid: true},
  x: {
    label: null,
    tickRotate: 90,
    domain: [
       "C4","C#4","D4","D#4", "Eb5",    
                    "E4","F4","F#4","G4",
                    "G#4", "A4", "A#4", "Bb4","B4",
                    "C5","C#5","Db5", "D5","Eb5",
                    "E5","F5","F#5","Gb5","G5","G#5",
                    "Ab5","A5", "A#5", "Bb5",
                    "B5","C6", "C#6", "Db6","D6", 
                    "D#6","Eb6","E6","F6","F#6","Gb6","G6"]
  },

  marks: [
    Plot.barY(data,
        Plot.groupX(
          {y: "count"}, 
          {x: "pitches"},
                               )),
    Plot.ruleY([0]),
  ]
})
